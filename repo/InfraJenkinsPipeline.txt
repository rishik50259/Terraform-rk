pipeline {
    agent any

    parameters {
        choice(name: 'ENV', choices: ['dev', 'stage', 'prod'], description: 'Select environment')
		choice(name: 'ACTION', choices: ['apply', 'destroy'], description: 'Select action')
    }
	

    environment {
        AWS_REGION = 'ap-south-1'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/rishik50259/Terraform-rk.git'
            }
        }
		
		stage('TFLint Check') {
		steps {
				sh """
				cd repo/infra/envs/${params.ENV}
				tflint
				"""
			}
		}
		
		

		
		stage('Checkov Security Scan') {
			steps {
					sh """
					cd repo/infra/envs/${params.ENV}
					/home/ubuntu/.local/bin/checkov -d . \
					--skip-check CKV_AWS_130,CKV2_AWS_12,CKV2_AWS_11
					"""
				}
		}


		
		
		stage ('Format Checking'){
			steps{
				sh "cd repo/infra/envs/${params.ENV} && terraform fmt -check"
			}
			
		}
		
		stage('Bootstrap Backend') {
			steps {
					sh """
					cd repo/bootstrap

					terraform init
					terraform apply -auto-approve
					
					"""
				}
		}

        stage('Terraform Init') {
            steps {
                sh "cd repo/infra/envs/${params.ENV} && terraform init -backend-config=../../backend/${params.ENV}.conf"
            }
        }

        stage('Terraform Validate') {
            steps {
                sh "cd repo/infra/envs/${params.ENV} && terraform validate"
            }
        }

        stage('Terraform Plan') {
            steps {
                sh "cd repo/infra/envs/${params.ENV} && terraform plan -out=tfplan"
            }
        }

        stage('Approval') {
            steps {
                script {
						if (params.ENV == 'prod') {
							input message: "🚨 Approve PRODUCTION deployment?"
						} else {
						echo "Auto-approved for ${params.ENV}"
						}
					}
			}
		}

        stage('Terraform Execute') {
            steps {
			
					script{
								if (params.ACTION ==  'apply'){
									sh "cd repo/infra/envs/${params.ENV} && terraform ${params.ACTION} tfplan"
								}else {
									sh "cd repo/infra/envs/${params.ENV} && terraform destroy -auto-approve"
								}
							}
				
            }
        }
    }
}
